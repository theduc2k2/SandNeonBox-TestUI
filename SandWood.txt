<script>
    /* --- FLOATING LEAVES ANIMATION --- */
    const bgCanvas = document.getElementById('bg-canvas');
    const bgCtx = bgCanvas.getContext('2d');
    let leaves = [];
    let bgWidth, bgHeight;

    function initBg() {
        resizeBg();
        window.addEventListener('resize', resizeBg);
        
        for(let i=0; i<15; i++) {
            leaves.push(createLeaf());
        }
        animateBg();
    }

    function resizeBg() {
        if (!bgCanvas.parentElement) return;
        bgWidth = bgCanvas.parentElement.offsetWidth;
        bgHeight = bgCanvas.parentElement.offsetHeight;
        bgCanvas.width = bgWidth;
        bgCanvas.height = bgHeight;
    }

    function createLeaf() {
        return {
            x: Math.random() * bgWidth,
            y: Math.random() * bgHeight - bgHeight,
            size: Math.random() * 15 + 10,
            speed: Math.random() * 1 + 0.5,
            swing: Math.random() * 2 - 1,
            rotation: Math.random() * Math.PI * 2,
            rotationSpeed: (Math.random() - 0.5) * 0.05,
            opacity: Math.random() * 0.3 + 0.2
        };
    }

    function animateBg() {
        bgCtx.clearRect(0, 0, bgWidth, bgHeight);
        
        leaves.forEach(leaf => {
            leaf.y += leaf.speed;
            leaf.x += Math.sin(leaf.y * 0.01) * leaf.swing;
            leaf.rotation += leaf.rotationSpeed;
            
            if (leaf.y > bgHeight) {
                leaf.y = -50;
                leaf.x = Math.random() * bgWidth;
            }

            bgCtx.save();
            bgCtx.translate(leaf.x, leaf.y);
            bgCtx.rotate(leaf.rotation);
            bgCtx.globalAlpha = leaf.opacity;
            
            // Draw simple leaf shape
            bgCtx.fillStyle = '#8D6E63';
            bgCtx.beginPath();
            bgCtx.ellipse(0, 0, leaf.size * 0.6, leaf.size, 0, 0, Math.PI * 2);
            bgCtx.fill();
            
            bgCtx.restore();
        });
        
        requestAnimationFrame(animateBg);
    }

    initBg();

    /* --- UI LOGIC --- */

        document.querySelectorAll('.nav-btn').forEach(el => el.classList.remove('active'));
        if(navEl) navEl.classList.add('active');
    }

    let particles = [], animationId;
    let score = 0;

    function resize() {
        const navEl = document.querySelector('.bottom-nav');
        const navHeight = navEl ? navEl.offsetHeight : 0;
        
        if (canvas.parentElement) {
            canvas.width = canvas.parentElement.offsetWidth;
            canvas.height = canvas.parentElement.offsetHeight - navHeight;
            CONFIG.w = canvas.width; 
            CONFIG.h = canvas.height;
        }
    }
    window.addEventListener('resize', resize); 
    resize();

    function startGame() {
        particles = []; 
        score = 0;
        document.getElementById('game-score').innerText = score;
        loop();
        setInterval(() => {
            const gameLayer = document.getElementById('game-layer');
            if(gameLayer && gameLayer.classList.contains('active')) {
                const randomColor = COLORS[Math.floor(Math.random() * COLORS.length)];
                spawnSand(Math.random() * CONFIG.w, 0, randomColor);
            }
        }, 60);
    }

    function spawnSand(x, y, color) {
        particles.push({x: x, y: y, color: color, sparkle: Math.random() > 0.7});
    }

    function loop() {
        animationId = requestAnimationFrame(loop);
        
        // Wooden gradient background
        const gradient = ctx.createLinearGradient(0, 0, 0, CONFIG.h);
        gradient.addColorStop(0, '#6D4C41');
        gradient.addColorStop(0.5, '#5D4037');
        gradient.addColorStop(1, '#4E342E');
        ctx.fillStyle = gradient;
        ctx.fillRect(0, 0, CONFIG.w, CONFIG.h);
        
        for(let i=0; i<particles.length; i++) {
            let p = particles[i];
            if(p.y < CONFIG.h - CONFIG.pSize) {
                p.y += CONFIG.pSize;
                if(Math.random() < 0.1) p.x += (Math.random() - 0.5) * 5; 
            }
            
            // Draw royal jewel particle
            ctx.save();
            ctx.translate(p.x + CONFIG.pSize/2, p.y + CONFIG.pSize/2);
            
            // Shadow
            ctx.fillStyle = 'rgba(0, 0, 0, 0.3)';
            ctx.beginPath();
            ctx.arc(2, 3, CONFIG.pSize/2, 0, Math.PI * 2);
            ctx.fill();
            
            // Gem body
            const gradient = ctx.createRadialGradient(0, -1, 0, 0, 0, CONFIG.pSize/2);
            gradient.addColorStop(0, p.color);
            gradient.addColorStop(0.6, p.color);
            gradient.addColorStop(1, 'rgba(0, 0, 0, 0.3)');
            ctx.fillStyle = gradient;
            ctx.beginPath();
            ctx.arc(0, 0, CONFIG.pSize/2, 0, Math.PI * 2);
            ctx.fill();
            
            // Highlight
            ctx.fillStyle = 'rgba(255, 255, 255, 0.6)';
            ctx.beginPath();
            ctx.arc(-1, -1, CONFIG.pSize/4, 0, Math.PI * 2);
            ctx.fill();
            
            // Sparkle
            if(p.sparkle && Math.random() > 0.95) {
                ctx.fillStyle = '#FFFFFF';
                ctx.fillRect(-1, -CONFIG.pSize, 2, CONFIG.pSize * 2);
                ctx.fillRect(-CONFIG.pSize, -1, CONFIG.pSize * 2, 2);
            }
            
            ctx.restore();
        }
        
        if(particles.length > 2800) {
            particles = [];
            document.getElementById('final-score').innerText = score;
            document.getElementById('game-over').classList.remove('hidden');
            cancelAnimationFrame(animationId);
        }
        score++;
        if(score % 10 === 0) document.getElementById('game-score').innerText = score;
    }
    
    canvas.addEventListener('touchmove', (e) => {
        e.preventDefault();
        let rect = canvas.getBoundingClientRect();
        let x = e.touches[0].clientX - rect.left;
        let y = e.touches[0].clientY - rect.top;
        spawnSand(x, y, COLORS[Math.floor(Math.random() * COLORS.length)]);
    }, {passive: false});

    canvas.addEventListener('mousemove', (e) => {
        if(e.buttons === 1) {
            let rect = canvas.getBoundingClientRect();
            let x = e.clientX - rect.left;
            let y = e.clientY - rect.top;
            spawnSand(x, y, COLORS[Math.floor(Math.random() * COLORS.length)]);
        }
    });

</script>
